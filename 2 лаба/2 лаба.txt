Функция ПреобразоватьФИО(ИсходноеФИО) Экспорт
    МассивЧастей = СтрРазделить(НРег(СтрЗаменить(ИсходноеФИО," ", " ")), " ");
    
    Фамилия = МассивЧастей[0]; 
    Инициалы = МассивЧастей[1]; 
    
    Фамилия = ВРег(Лев(Фамилия,1)) + НРег(Сред(Фамилия,2));
    
    МассивИнициалов = СтрРазделить(Инициалы,".");
    ИмяИнициала = ВРег(МассивИнициалов[0]);
    ОтчИнициала = ВРег(МассивИнициалов[1]);
    
    Результат = Фамилия + " " + ИмяИнициала + "." + ОтчИнициала + ".";
    Возврат Результат;
КонецФункции
